// Vende itens, em Prontera, usando NossaCoin
// Troca NossaCoin em grandes quantidades!

prontera,151,187,5	script	Nossa Loja	72,{
	mes "[Nossa Loja]";
	mes "Olá, escolha sua opção:";
	switch(select("-Trocar NossaCoin.:-Comprar Itens.:-Outras opções.")) {
		case 1: // Comprar NossaCoin
			clear;
			mes "[Nossa Loja]";
			mes "Informe a quantidade de NossaCoins para trocar.";
			mes "Você possui " + #NOSSACOIN + " NossaCoins";
			mes "Caso deseja cancelar, informe 0.";
			input .@Value;
			if(#NOSSACOIN >= .@Value){
				set #NOSSACOIN, #NOSSACOIN - .@Value;
				if (.@Value <= 0) {
					clear;
					mes "[Nossa Loja]";
					mes "Valor incorreto!";
					mes "Você possui apenas " + #NOSSACOIN + " NossaCoins";
					close;
				} else {
				getitem 840095,.@Value;
				clear;
				mes "[Nossa Loja]";
				mes "Você trocou " + .@Value + " NossaCoins, volte sempre!";
				mes "Você tem " + #NOSSACOIN + " NossaCoins restantes.";
				close;
				}
			}
		case 2: // Comprar itens
			clear;
			mes "[Nossa Loja]";
			//mes "Em construção, volte em breve!";
			mes "Aqui está!";
			close2;
			callshop "nossaLoja",1;
			end;
		case 3: // Outras opcoes
			clear;
			mes "[Nossa Loja]";
			mes "Posso trocar grandes quantias de NossaCoin!";
			mes "Caso deseja cancelar, informe 0.";
			switch(select("-Ok, vamos lá!:-Não, obrigado...")){
				case 1: //Trocar nossa coin
				input .@Value;
				if(countitem(840095) >= .@Value){
				set #NOSSACOIN, #NOSSACOIN + .@Value;
				delitem 840095,.@Value;
				clear;
				mes "[Nossa Loja]";
				mes "Você trocou " + .@Value + " NossaCoins, volte sempre!";
				mes "Você tem " + #NOSSACOIN + " NossaCoins na conta, e " + countitem(840095) + " no inventário!";
				close;
				} else {
					clear;
					mes "[Nossa Loja]";
					mes "Você não tem " + .@Value + " NossaCoins para trocar...";
					mes "Você tem " + countitem(840095) + " NossaCoins.";
					close;
				}
			}
	}
}
